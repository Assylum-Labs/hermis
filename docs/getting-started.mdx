## Installation

Choose the package that best fits your needs:

<Tabs>
  <Tab title="React">
    ```bash
    npm install @hermis/solana-headless-react @solana/web3.js
    ```

    This package includes everything you need for React applications, including hooks and context providers.
  </Tab>
  <Tab title="Core">
    ```bash
    npm install @hermis/solana-headless-core @solana/web3.js
    ```

    The core package provides framework-agnostic wallet functionality that works anywhere.
  </Tab>
  <Tab title="Adapter Base">
    ```bash
    npm install @hermis/solana-headless-adapter-base @solana/web3.js
    ```

    Use this package when you need to create custom wallet adapters or extend existing ones.
  </Tab>
</Tabs>

## Quick Setup

Here's a minimal example to get you started:

<CodeGroup>
```tsx React
import { HermisProvider, useHermis } from '@hermis/solana-headless-react';

function App() {
  return (
    <HermisProvider
      network="mainnet-beta"
      autoConnect={false}
    >
      <WalletButton />
    </HermisProvider>
  );
}

function WalletButton() {
  const { connect, disconnect, connected, publicKey } = useHermis();

  return (
    <button onClick={connected ? disconnect : connect}>
      {connected ? `Connected: ${publicKey?.toBase58()}` : 'Connect Wallet'}
    </button>
  );
}
```

```javascript Vanilla JS
import { HermisClient } from '@hermis/solana-headless-core';
import { Connection, clusterApiUrl } from '@solana/web3.js';

// Initialize the client
const connection = new Connection(clusterApiUrl('mainnet-beta'));
const hermis = new HermisClient({ connection });

// Connect to a wallet
async function connectWallet() {
  try {
    await hermis.connect();
    console.log('Connected:', hermis.publicKey?.toBase58());
  } catch (error) {
    console.error('Failed to connect:', error);
  }
}

// Disconnect wallet
async function disconnectWallet() {
  await hermis.disconnect();
  console.log('Disconnected');
}
```

```typescript TypeScript
import { HermisClient, type WalletAdapter } from '@hermis/solana-headless-core';
import { Connection, PublicKey } from '@solana/web3.js';

interface AppConfig {
  network: 'mainnet-beta' | 'devnet' | 'testnet';
  autoConnect: boolean;
}

class WalletManager {
  private hermis: HermisClient;

  constructor(config: AppConfig) {
    const connection = new Connection(
      config.network === 'mainnet-beta'
        ? 'https://api.mainnet-beta.solana.com'
        : `https://api.${config.network}.solana.com`
    );

    this.hermis = new HermisClient({
      connection,
      autoConnect: config.autoConnect,
    });
  }

  async connect(): Promise<PublicKey | null> {
    await this.hermis.connect();
    return this.hermis.publicKey;
  }

  async disconnect(): Promise<void> {
    await this.hermis.disconnect();
  }

  get publicKey(): PublicKey | null {
    return this.hermis.publicKey;
  }

  get connected(): boolean {
    return this.hermis.connected;
  }
}

export { WalletManager };
```
</CodeGroup>

## Configuration Options

Configure the SDK to match your application's needs:

<ParamField path="network" type="string" default="mainnet-beta">
  The Solana network to connect to: `mainnet-beta`, `devnet`, or `testnet`
</ParamField>

<ParamField path="autoConnect" type="boolean" default="false">
  Automatically connect to a previously connected wallet on page load
</ParamField>

<ParamField path="commitment" type="string" default="confirmed">
  The commitment level for transactions: `processed`, `confirmed`, or `finalized`
</ParamField>

<ParamField path="wallets" type="WalletAdapter[]">
  Array of wallet adapters to support. If not provided, all available wallets are supported.
</ParamField>
