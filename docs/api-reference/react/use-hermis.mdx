## useHermis()

Primary hook combining wallet and connection functionality.

```typescript
import { useHermis } from '@hermis/solana-headless-react';

function MyComponent() {
  const { connect, disconnect, connected, publicKey, connection } = useHermis();

  return (
    <button onClick={connected ? disconnect : connect}>
      {connected ? `Connected: ${publicKey?.toBase58()}` : 'Connect'}
    </button>
  );
}
```

## Return Values

<ResponseField name="connect" type="() => Promise<void>">
  Connect to wallet
</ResponseField>

<ResponseField name="disconnect" type="() => Promise<void>">
  Disconnect from wallet
</ResponseField>

<ResponseField name="connected" type="boolean">
  Connection status
</ResponseField>

<ResponseField name="connecting" type="boolean">
  Connection in progress
</ResponseField>

<ResponseField name="publicKey" type="PublicKey | null">
  Wallet public key
</ResponseField>

<ResponseField name="connection" type="Connection">
  Solana RPC connection
</ResponseField>

<ResponseField name="sendTransaction" type="(tx: Transaction, connection: Connection) => Promise<string>">
  Send a transaction
</ResponseField>

<ResponseField name="signTransaction" type="(tx: Transaction) => Promise<Transaction>">
  Sign a transaction
</ResponseField>

<ResponseField name="signMessage" type="(msg: Uint8Array) => Promise<Uint8Array>">
  Sign a message
</ResponseField>
